let cart = [];

function renderEquipment(){

    const grid = document.getElementById("equipmentGrid");

    grid.innerHTML = "";

    equipmentData.forEach((item,index)=>{

        grid.innerHTML += `
        
        <div class="equipment-card">

            <div class="equipment-image"
            style="background-image:url('${item.image}')"></div>

            <div class="equipment-body">

                <h3>${item.name}</h3>

                <p>หมวดหมู่: ${item.category}</p>
                <p>คงเหลือ: ${item.quantity}</p>

                <span class="status available">${item.status}</span>

                <button class="btn"
                style="margin-top:15px"
                onclick="addToCart(${index})">
                เพิ่มลงตะกร้า
                </button>

            </div>

        </div>

        `;

    });

}

function addToCart(index){

    cart.push(equipmentData[index]);

    showToast("เพิ่มอุปกรณ์ลงตะกร้าแล้ว");

}

function openCart(){

    alert(
`🛒 จำนวนสินค้าในตะกร้า: ${cart.length}

ระบบจะ:
- สร้างเลขรายการอัตโนมัติ
- ลดจำนวนอุปกรณ์อัตโนมัติ
- เปลี่ยนสถานะเป็นรออนุมัติ`
);

}

function showToast(message){

    const toast = document.getElementById("toast");

    toast.innerText = message;
    toast.style.display = "block";

    setTimeout(()=>{
        toast.style.display = "none";
    },3000);

}

function loginStudent(){

    const name = document.getElementById("studentName").value;
    const classroom = document.getElementById("studentClass").value;
    const room = document.getElementById("studentRoom").value;
    const no = document.getElementById("studentNo").value;

    if(!name || !room || !no){

        alert("กรุณากรอกข้อมูลให้ครบ");
        return;

    }

    const studentData = {
        name,
        classroom,
        room,
        no
    };

    localStorage.setItem("studentSession",JSON.stringify(studentData));

    document.getElementById("loginPage").style.display = "none";
    document.getElementById("app").style.display = "block";

    showToast("เข้าสู่ระบบสำเร็จ");

}

function logout(){

    localStorage.removeItem("studentSession");

    location.reload();

}

function adminLogin(){

    const pass = prompt("กรอกรหัสแอดมิน");

    if(pass === "14150"){

        alert("เข้าสู่ระบบแอดมินสำเร็จ");

    }else{

        alert("รหัสไม่ถูกต้อง");

    }

}

window.onload = ()=>{

    renderEquipment();

    const session = localStorage.getItem("studentSession");

    if(session){

        document.getElementById("loginPage").style.display = "none";
        document.getElementById("app").style.display = "block";

    }

};
